[docs]    def load(self, device: str):
        """
        Load user-selected task-specific model

        Args:
            device (str): device information

        Returns:
            object: User-selected task-specific model

        """
        from pororo.tasks import PororoAspectFactory, PororoSentimentFactory

        aspect = PororoAspectFactory(
            task="aspect",
            model="brainbert.base.ko.aspect.v2",
            lang="ko",
        ).load(device)

        sentiment = PororoSentimentFactory(
            task="sentiment",
            model="brainbert.base.ko.general_review",
            lang="ko",
        ).load(device)
        return PororoBertABSA(aspect, sentiment, self.config)

    def merge(labels: List):
        """
        Merge successive label to build more general span

        Args:
            labels (list): predicted label list

        Returns:
            list: merged result

        """
        result = []
        prev_aspect = None

        if len(labels) == 1:
            return labels

        for label in labels:
            curr_aspect = label[1]
            if (curr_aspect == prev_aspect) and (curr_aspect is not None):
                result[-1][0] += " " + label[0]
            else:
                result.append(label)
                prev_aspect = label[1]
        return result



[docs]    def label_sentiment(self, labels, sent: str, threshold: float):
        """
        Label sentiment according to pre-defined threshold

        Args:
            labels (list): predicted labels list
            sent (str): input sentence
            threshold (float): threshold deciding sentiment label

        Returns:
            list: sentiment appended result list

        """

        def predict_sentiment(text: str, threshold: float):
            res = self.sentiment(text, show_probs=True)
            if max(res.values()) > threshold:
                return max(res, key=res.get).capitalize()
            return "Neutral"

        sent = unicodedata.normalize("NFKC", sent)
        result = []
        for label in labels:

            if not label[1]:
                continue

            start = sent.find(label[0])
            end = start + len(label[0])
            result.append({
                "mention": label[0],
                "aspect": label[1],
                "sentiment": predict_sentiment(label[0], threshold),
                "start": start,
                "end": end,
            })
            start += len(label[0])

        return result



[docs]    def predict(
        self,
        sent: str,
        aspect_threshold: float,
        polarity_threshold: float,
        do_merge: bool,
    ) -> Mapping:
        """
        Conduct sentiment analysis

        Args:
            sent: (str) sentence to be sentiment analyzed
            aspect_threshold: (float) threshold to determine whether it is specific aspect or not.
            polarity_threshold: (float) threshold to determine whether it is positive or negative.
            do_merge: (bool) if True, ‘satisfaction’ aspect is merged with other aspects

        Returns:
            str: predicted sentence with aspect and polarity

        """
        sent = sent.replace("\n", "")
        labels = self.aspect(sent, threshold=aspect_threshold)

        if do_merge:
            labels = self.merge(labels)

        result = self.label_sentiment(
            labels,
            sent,
            threshold=polarity_threshold,
        )
        return result


    def __call__(
        self,
        text: str,
        aspect_threshold: float = 0.6,
        polarity_threshold: float = 0.9,
        do_merge: bool = True,
    ):
        assert isinstance(text, str), "Input text should be string type"

        return self.predict(
            text,
            aspect_threshold,
            polarity_threshold,
            do_merge,
        )
