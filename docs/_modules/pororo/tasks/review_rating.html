from typing import Optional

from pororo.tasks.utils.base import PororoFactoryBase, PororoSimpleBase



[docs]class PororoReviewFactory(PororoFactoryBase):
    """
    Classification based review rating using Review Corpus

    Korean (`brainbert-base`)
        Accuracy = 60.18

    Korean (`brainbert-large`)
        Accuracy = 59.87

    Examples:
        >>> review = Pororo(task="review", lang="ko")
        >>> review("이거 진짜 돈 주고 사는 사람이 있음?")
        1
        >>> review("애매한데 쓸만하긴 한 것 같네요")
        3
        >>> review("가슴이 웅장해진다. 내가 알고있던 세제가 맞나?")
        5
        >>> review("싼거 같아서 샀는데 별로인거 같아")
        2
        >>> review("올해 구입한 것중에서 만족도가 제일 높아")
        5

    """

    def __init__(self, task: str, lang: str, model: Optional[str]):
        super().__init__(task, lang, model)

    @staticmethod
    def get_available_langs():
        return ["ko"]

    @staticmethod
    def get_available_models():
        return {"ko": ["brainbert.base.ko.review", "brainbert.large.ko.review"]}


[docs]    def load(self, device: str):
        """
        Load user-selected task-specific model

        Args:
            device (str): device information

        Returns:
            object: User-selected task-specific model

        """
        if "brainbert" in self.config.n_model:
            from pororo.models.brainbert import BrainRobertaModel

            model = (BrainRobertaModel.load_model(
                f"bert/{self.config.n_model}",
                self.config.lang,
            ).eval().to(device))
            return PororoBertReview(model, self.config)

[docs]    def predict(self, sent: str) -> int:
        """
        Conduct review rating scaled from 1 to 5

        Args:
            sent: (str) sentence to be rated

        Returns:
            int: rating score scaled from 1 to 5

        """
        return int(self._model.predict_output(sent))
