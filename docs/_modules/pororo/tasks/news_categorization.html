[docs]    def load(self, device: str):
        """
        Load user-selected task-specific model

        Args:
            device (str): device information

        Returns:
            object: User-selected task-specific model

        """
        if "brainbert" in self.config.n_model:
            from pororo.models.brainbert import BrainRobertaModel

            model = (BrainRobertaModel.load_model(
                f"bert/{self.config.n_model}",
                self.config.lang,
            ).eval().to(device))
            return PororoBertNews(model, self.config)

[docs]    def predict(self, sent: str):
        """
        Conduct news categorization

        Args:
            sent: (str) sentence to be categorized into specific news category

        Returns:
            str: categorized news class - Categories: living_culture, economy, politics, world, opinion, sports, society, IT_science

        """
        sent = sent.replace("\n", " ||| ")
        return self._model.predict_output(sent).capitalize()
