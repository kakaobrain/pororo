[docs]class PororoTtsFactory(PororoFactoryBase):
    """
    Synthesis text to speech using trained model.
    Output audio's sample rate is 22050.

    Multi (`mTacotron`)

        - dataset: TBU
        - metric: TBU

    Args:
        text (str): text for speech synthesis
        lang (str): text's language Ex) how are you?: en, 안녕하세요.: ko
        speaker (str): designate a speaker such as ko, en, zh etc.. (default: lang)

    Returns:
        ndarray: waveform of speech signal

    Examples:
        >>> import IPython
        >>> from IPython.display import Audio
        >>> model = Pororo('tts', lang='multi')
        >>> wave = model('how are you?', lang='en', speaker='en')
        >>> IPython.display.display(IPython.display.Audio(data=wave, rate=22050))

        >>> model = Pororo('tts', lang='multi')
        >>> wave = model('저는 미국 사람이에요.', lang='ko', speaker='en')
        >>> IPython.display.display(IPython.display.Audio(data=wave, rate=22050))

    Notes:
        Currently 10 languages supports.
        Supported Languages: English, Korean, Japanese, Chinese, Jejueo, Dutch, German, Spanish, French, Russian
        This task can designate a speaker such as ko, en, zh etc.

    """

    def __init__(self, task: str, lang: str, model: Optional[str]):
        super().__init__(task, lang, model)

[docs]    def load(self, device: str):
        """
        Load user-selected task-specific model

        Args:
            device (str): device information

        Returns:
            object: User-selected task-specific model

        """
        if self.config.n_model == "mTacotron":
            from pororo.models.mTTS.synthesizer import (
                MultilingualSpeechSynthesizer,
            )
            from pororo.models.mTTS.utils.numerical_pinyin_converter import (
                convert_from_numerical_pinyin,
            )
            from pororo.models.mTTS.utils.text import jejueo_romanize, romanize

            tacotron_base_path = download_or_load(
                "misc/mTacotron-base",
                self.config.lang,
            )
            tacotron_large_path = download_or_load(
                "misc/mTacotron",
                self.config.lang,
            )
            wavernn_path = download_or_load(
                "misc/wavernn.pyt",
                self.config.lang,
            )
            base_model = MultilingualSpeechSynthesizer(
                tacotron_base_path,
                wavernn_path,
                device,
                self.config.lang,
            )
            large_model = MultilingualSpeechSynthesizer(
                tacotron_large_path,
                wavernn_path,
                device,
                self.config.lang,
            )
            return PororoTTS(
                base_model,
                large_model,
                device,
                romanize,
                jejueo_romanize,
                convert_from_numerical_pinyin,
                self.config,
            )

    def _preprocess(
        self,
        text: str,
        lang: str = "en",
        speaker: str = None,
    ) -> str:
        """
        Pre-process text for TTS format

        Args:
            text (str): text for tts
            lang (str): text language
            speaker (speaker): designation of speaker

        Returns:
            str: pre-processed text

        """
        if lang == "ko":
            text = self.romanize(text)
        elif lang == "ja":
            if self.g2p_ja is None:
                self.load_g2p_ja()
            text = self.g2p_ja(text)
        elif lang == "zh":
            if self.g2p_zh is None:
                self.load_g2p_zh()
            text = self.g2p_zh(text).replace("   ", " ")
            text = self.convert_from_numerical_pinyin(text)
        elif lang == "je":
            text = self.jejueo_romanize(text)
        return f"{text}|{speaker}|{speaker}"


[docs]    def predict(self, text: str, model="base") -> ndarray:
        """
        Conduct speech synthesis on given text

        Args:
            text (str): text for tts
            model (str): designate model (base or large)

        Returns:
             ndarray: waveform of speech signal

        """
        if model == "base":
            return self._base_model.predict(text)
        return self._large_model.predict(text)


    def __call__(self, text: str, lang: str = "en", speaker: str = None):
        model = "base"

        if speaker is None:
            speaker = lang

        if lang == "je" or speaker == "je":
            model = "large"

        assert (
            speaker in self.speaker.keys()
        ), f"Unsupported speaker: {speaker}\nSupported speaker: {self.speaker.keys()}"
        text = self._preprocess(text, lang, self.speaker[speaker])
        return self.predict(text, model)
