[docs]    def load(self, device: str):
        """
        Load user-selected task-specific model

        Args:
            device (str): device information

        Returns:
            object: User-selected task-specific model

        """
        if "charbert" in self.config.n_model:
            try:
                import mecab
            except ModuleNotFoundError as error:
                raise error.__class__(
                    "Please install python-mecab-ko with: `pip install python-mecab-ko`"
                )

            from pororo.models.brainbert import CharBrainRobertaModel
            from pororo.tasks import PororoNerFactory

            pos_tagger = mecab.MeCab()
            ner_tagger = PororoNerFactory(
                task="ner",
                model="charbert.base.ko.ner",
                lang=self.config.lang,
            ).load(device)

            model = (CharBrainRobertaModel.load_model(
                f"bert/{self.config.n_model}",
                self.config.lang,
            ).eval().to(device))
            return PororoBertCoref(model, pos_tagger, ner_tagger, self.config)

    def __init__(self, model, pos_tagger, ner_tagger, config):
        super().__init__(config)
        self._model = model
        self._pos_tagger = pos_tagger
        self._ner_tagger = ner_tagger
        self._person_lst = ["그", "그녀", "그들"]
        self._loc_lst = ["이곳", "그곳"]


[docs]    def get_proper_noun(self, sent: str):
        """
        Get Proper Noun from input sentence

        Args:
            sent (str): sentence to be checked whether it contains proper noun

        Returns:
            List[Tuple[str, Tuple[int, int]]]: proper noun list

        """
        sent = sent.strip()
        text_ptr = 0

        pairs = list()
        for unit in self._pos_tagger.pos(sent):
            token = unit[0]
            label = unit[1]
            if sent[text_ptr] == " ":
                while sent[text_ptr] == " ":
                    text_ptr += 1
                pairs.append((" ", "SPACE"))
            pairs.append((token, label))
            text_ptr += len(token)

        idx = 0
        result = list()
        for pair in pairs:
            token, label = pair

            if label == "NP":
                if (token in self._person_lst) or (token in self._loc_lst):
                    result.append((token, (idx, idx + len(token))))
            idx += len(token)

        return result



[docs]    def postprocess(self, tokens: List[str]) -> str:
        """
        Postprocess predicted result

        Args:
            token (List[Tuple[str, int]]): predicted list from coreference resolution model

        Returns:
            str: coreferenced noun string

        """
        result = ""
        for token in tokens:
            if token[-1] == "1":
                result += token[0]
        result = result.replace("<PS>", "")
        result = result.replace("</PS>", "")
        result = result.replace("<LCOG>", "")
        result = result.replace("</LCOG>", "")
        return result.replace("▁", " ")



[docs]    def predict(self, sent: str) -> List[Tuple[str, int, str]]:
        """
        Conduct Coreference Resolution using POS, NER and Coref module

        Args:
            sent: (str) sentence to be resolved coreference

        Returns:
            List[Tuple(str, int, str)]: predicted coreference and its corresponding proper noun

        """
        # STEP 1. search for NPs using the pos model of pororo
        # & check if it is personal pronoun or not.
        cands = self.get_proper_noun(sent)

        if not cands:
            return "There is no <Proper Noun> coreferenced"

        # STEP 2. apply ner with candidate
        ner_tags = self._ner_tagger(sent)

        # STEP 3. switch the format to xml & character tokenize
        result = list()
        for cand in cands:
            cand_np, cand_idx = cand
            idx = 0
            xml = ""
            for ner_tag in ner_tags:
                token, tag = ner_tag

                start_idx = idx
                end_idx = idx + len(token)

                if (cand_idx[0] <= start_idx) and (cand_idx[-1] <= end_idx):
                    if cand_np in self._person_lst:
                        token = token.replace(cand_np, "<PS/>")
                    elif cand_np in self._loc_lst:
                        token = token.replace(cand_np, "<LCOG/>")

                if tag in ["PERSON", "LOCATION", "ORGANIZATION"]:
                    if tag in ["LOCATION", "ORGANIZATION"]:
                        tag = "LCOG"
                    if tag == "PERSON":
                        tag = "PS"
                    dummy = f"<{tag}> {' '.join([char if char != ' ' else '▁' for char in token])} </{tag}> "
                    xml += dummy
                elif token == " ":
                    xml += "▁ "
                elif "<PS/>" in token:
                    xml += f"<PS/> {' '.join([char for char in token[5:]])}"
                elif "<LCOG/>" in token:
                    xml += f"<LCOG/> {' '.join([char for char in token[7:]])}"
                else:
                    xml += " ".join([char for char in token]) + " "
                idx += len(token)
            xml = xml.strip()
            prediction = self.postprocess(self._model.predict_coref(xml))
            if prediction:
                result.append((cand_np, cand_idx[0], prediction))
            else:
                result.append((cand_np, cand_idx[0], "N/A"))

        return result
