[docs]    def load(self, device: str):
        """
        Load user-selected task-specific model

        Args:
            device (str): device information

        Returns:
            object: User-selected task-specific model

        """
        if "posbert" in self.config.n_model:
            from pororo.models.brainbert import RobertaSegmentModel

            model = (RobertaSegmentModel.load_model(
                f"bert/{self.config.n_model}",
                self.config.lang,
            ).eval().to(device))

            tagger = PororoPosFactory(
                task="pos",
                model="mecab-ko",
                lang=self.config.lang,
            ).load(device)

            return PororoSegmentBertDP(model, tagger, self.config)

        if "charbert" in self.config.n_model:
            from pororo.models.brainbert import CharBrainRobertaModel

            model = (CharBrainRobertaModel.load_model(
                f"bert/{self.config.n_model}",
                self.config.lang,
            ).eval().to(device))

            tagger = PororoPosFactory(
                task="pos",
                model="mecab-ko",
                lang=self.config.lang,
            ).load(device)

            return PororoBertDP(model, tagger, self.config)

[docs]    def preprocess(self, sent: str) -> str:
        """
        Preprocess dependency parsing input

        Args:
            sent (str): input sentence to be preprocessed

        Returns:
            str: preprocessed input sentence with pos tag

        """
        pairs = self._tagger(sent, return_surface=True)
        tokens = ["<s>", "▃"
                 ] + [pair[0] if pair[0] != " " else "▃" for pair in pairs]
        tags = [
            pair[1] if pair[0] != " " else pairs[i + 1][1]
            for i, pair in enumerate(pairs)
        ]
        prefix = ["XX", tags[0]]
        tags = prefix + tags

        res_tags = []
        for tag in tags:
            if "+" in tag:
                tag = tag[:tag.find("+")]
            res_tags.append(tag)
        return tokens, res_tags



[docs]    def postprocess(
        self,
        ori: str,
        tokens: List[str],
        heads: List[str],
        labels: List[str],
    ):
        """
        Postprocess dependency parsing output

        Args:
            ori (sent): original sentence
            heads (List[str]): dependency heads generated by model
            labels (List[str]): tag labels generated by model

        Returns:
            List[Tuple[int, str, int, str]]: token index, token label, token head and its relation

        """
        eojeols = ori.split()

        indices = [i for i, token in enumerate(tokens) if token == "▃"]
        real_heads = [head for i, head in enumerate(heads) if i in indices]
        real_labels = [label for i, label in enumerate(labels) if i in indices]

        result = []
        for i, (head, label,
                eojeol) in enumerate(zip(
                    real_heads,
                    real_labels,
                    eojeols,
                )):
            curr = i + 1

            try:
                head_eojeol = indices.index(head) + 1
            except:
                head_eojeol = -1

            if head_eojeol == curr:
                head_eojeol = -1

            result.append((curr, eojeol, head_eojeol, label))
        return result



[docs]    def predict(self, sent: str):
        """
        Conduct dependency parsing

        Args:
            sent: (str) sentence to be parsed dependency

        Returns:
            List[Tuple[int, str, int, str]]: token index, token label, token head and its relation

        """
        tokens, tags = self.preprocess(sent)
        heads, labels = self._model.predict_dependency(tokens, tags)
        heads = [int(head) - 1 for head in heads]  # due to default <s> token
        return self.postprocess(sent, tokens, heads, labels)

[docs]    def preprocess(self, sent: str) -> str:
        """
        Preprocess dependency parsing input

        Args:
            sent (str): input sentence to be preprocessed

        Returns:
            str: preprocessed input sentence with pos tag

        """
        pairs = self._tagger(sent)

        res = str()
        for pair in pairs:
            if pair[1] == "SPACE":
                res = res[:-2]
                res += " ▁ "
                continue
            chars = " ".join([c for c in pair[0]])
            res += f"{chars} / {pair[1]} + "
        return res[:-2]



[docs]    def postprocess(
        self,
        ori: str,
        dp_labels: List[str],
        tag_labels: List[str],
    ):
        """
        Postprocess dependency parsing output

        Args:
            ori (sent): original sentence
            dp_labels (List[str]): dependency labels generated by model
            tag_labels (List[str]): tag labels generated by model

        Returns:
            List[Tuple[int, str, int, str]]: token index, token label, token head and its relation

        """
        result = list()
        tokens = ori.split()

        assert dp_labels.count("0") == tag_labels.count(
            "SPACE"), "Cannot use the input sentence"

        for i, token in enumerate(tokens):
            head = int(dp_labels[0])
            if head == i + 1:
                head = -1
            result.append((i + 1, token, head, tag_labels[0]))
            if i < (len(tokens) - 1):
                dp_labels = dp_labels[dp_labels.index("0") + 1:]
                tag_labels = tag_labels[tag_labels.index("SPACE") + 1:]

        return result

        """

            sent: (str) sentence to be parsed dependency

        Returns:
            List[Tuple[int, str, int, str]]: token index, token label, token head and its relation

        Examples:
            >>> dp = Pororo(task="dep_parse", lang="ko")
            >>> dp("분위기도 좋고 음식도 맛있었어요. 한 시간 기다렸어요.")
            [(1, '분위기도', 2, 'NP_SBJ'), (2, '좋고', 4, 'VP'), (3, '음식도', 4, 'NP_SBJ'), (4, '맛있었어요.', 7, 'VP'), (5, '한', 6, 'NP'), (6, '시간', 7, 'NP_OBJ'), (7, '기다렸어요.', -1, 'VP')]
            >>> dp("한시간 기다렸어요.")
            [(1, '한시간', 2, 'NP_AJT'), (2, '기다렸어요.', -1, 'VP')]

        """
        preprocessed = self.preprocess(sent)
        dp_labels, tag_labels = self._model.predict_dependency(preprocessed)
        return self.postprocess(sent, dp_labels, tag_labels)
