"""Clause Tokenization related modeling class"""

from typing import Optional, List

from pororo.tasks.utils.base import PororoFactoryBase, PororoSimpleBase


class PororoClauseFactory(PororoFactoryBase):

    def __init__(self, task: str, lang: str, model: Optional[str]):
        super().__init__(task, lang, model)

    @staticmethod
    def get_available_langs():
        return ["ko"]

    @staticmethod
    def get_available_models():
        return {"ko": ["charbert.base.ko.clause"]}

    def load(self, device: str):
        """
        Load user-selected task-specific model

        Args:
            device (str): device information

        Returns:
            object: User-selected task-specific model

        """
        if "charbert" in self.config.n_model:
            from pororo.models.brainbert import CharBrainRobertaModel
            model = CharBrainRobertaModel.load_model(
                f"bert/{self.config.n_model}",
                self.config.lang,
            ).eval().to(device)
            return PororoBertClause(model, self.config)

[docs]    def postprocess(self, tokens: List[str]) -> List[str]:
        """
        Post-process clause tokenized tokens

        Args:
            tokens (List[str]): each token and token's predicted label

        Returns:
            List[str]: Post-processed clause list

        """
        results = list()
        clause = str()

        for pair in tokens:
            token, label = pair
            clause += token

            if label == "1":
                results.append(clause)
                clause = str()

        results.append(clause)
        return [clause.replace("▁", " ").strip() for clause in results]



[docs]    def predict(self, sent: str) -> List[str]:
        """
        Conduct clause tokenization

        Args:
            sent: (str) sentence to be tokenized

        Returns:
            List[str]: tokenized clause list

        Examples:
            >>> clause = Pororo(task="clause", lang="ko")
            >>> clause("꽤 맘에 들었어요. 겉에서 봤을땐 허름?했는데 맛도 있고, 괜찮아요")
            ['꽤 맘에 들었어요.', '겉에서 봤을땐 허름?했는데', '맛도 있고, 괜찮아요']
            >>> clause("분위기도 좋고 음식도 맛있었어요")
            ['분위기도 좋고', '음식도 맛있었어요']

        """
        if isinstance(sent, str):
            sent = [sent]

        result = self._model.predict_clause(sent)

        li_result = []
        for r in result:
            li_result.append(self.postprocess(r))

        return li_result if len(li_result) > 1 else li_result[0]
