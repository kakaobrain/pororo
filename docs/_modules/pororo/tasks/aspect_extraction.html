[docs]    def load(self, device: str):
        """
        Load user-selected task-specific model

        Args:
            device (str): device information

        Returns:
            object: User-selected task-specific model

        """
        if "brainbert" in self.config.n_model:
            from pororo.models.brainbert import BrainRobertaModel
            from pororo.tasks import (
                PororoPosFactory,
                PororoTokenizationFactory,
            )

            model = (BrainRobertaModel.load_model(
                f"bert/{self.config.n_model}",
                self.config.lang,
            ).eval().to(device))

            clause = PororoTokenizationFactory(
                task="tokenization",
                lang="ko",
                model="clause",
            ).load(device)

            pos = PororoPosFactory(
                task="pos",
                lang="ko",
                model="mecab-ko",
            ).load(device)

            return PororoBertAspect(model, clause, pos, self.config)

    def __init__(self, model, clause, pos, config):
        super().__init__(config)
        self._model = model
        self._clause = clause
        self._pos = pos

        self.aspect_dict = {
            "배송": ["배송", "택배", "배달", "빠르", "느리"],
            "가격": ["가격", "금액", "저렴", "가성", "비싸", "싸"],
            "품질": ["품질", "질", "퀄리티", "퀄러티", "퀄", "튼튼"],
            "만족도": ["만족", "좋", "마음", "별로"],
        }

        self._label_fn = {
            "0": "배송",
            "1": "가격",
            "2": "품질",
            "3": "만족도",
            "4": None
        }


[docs]    def apply_baseline(self, text: str):
        """
        Apply rule-based baseline to extract aspect

        Args:
            text (str): input sentence to be analyzed

        Returns:
            str or None: return extracted aspect name or None

        """
        tags = self._pos(text)
        for tag in tags:
            token, _ = tag
            for key, value in self.aspect_dict.items():
                if token in value:
                    return key
        return None



[docs]    def predict(self, sent: str, threshold: float, show_probs: bool = False):
        """
        Conduct Aspect Extraction

        Args:
            sent (str): sentence to be analyzed
            threshold (float): base threshold score
            show_probs (bool): whether to show probability

        Returns:
            List: predicted aspect list

        """
        result = []

        for segment in self._clause(sent):
            # apply rule-based baseline,
            # if caught in this logic continue to next segment
            aspect = self.apply_baseline(segment)
            if aspect:
                label = [segment, aspect]
                if show_probs:
                    label.append(1.0)
                result.append(label)
                continue

            # apply pipeline module
            aspect_pred = self._model.predict_output(segment, show_probs=True)
            aspect = max(
                aspect_pred,
                key=aspect_pred.get,
            )  # get aspect with maximum prob
            prob = round(aspect_pred[aspect], 3)

            label = [segment, None]
            if prob > threshold:
                label[-1] = self._label_fn[aspect]

            if show_probs:
                label.append(prob)
            result.append(label)

        return result


    def __call__(
        self,
        text: str,
        threshold: float = 0.65,
        show_probs: bool = False,
    ):
        assert isinstance(text, str), "Input text should be string type"

        text = self._normalize(text)

        return self.predict(text, threshold, show_probs)
