[docs]    def load(self, device: str):
        """
        Load user-selected task-specific model

        Args:
            device (str): device information

        Returns:
            object: User-selected task-specific model

        """
        if self.config.n_model == "vad":
            try:
                import librosa
            except ModuleNotFoundError as error:
                raise error.__class__(
                    "Please install librosa with: `pip install librosa`")

            from pororo.models.vad import VoiceActivityDetection

            model_path = download_or_load(
                "misc/vad.pt",
                self.config.lang,
            )
            model = VoiceActivityDetection(
                model_path=model_path,
                device=device,
            )

            return PororoVAD(
                model=model,
                config=self.config,
            )
